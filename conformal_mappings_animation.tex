\newcommand{\PgfmathsetmacroFPU}[2]{\begingroup%
\pgfkeys{/pgf/fpu,/pgf/fpu/output format=fixed}%
\pgfmathsetmacro{#1}{#2}%
\pgfmathsmuggle#1\endgroup}%
\def\conformaltransformation#1{%
\PgfmathsetmacroFPU{\myphase}{atan2(\the\pgf@y,\the\pgf@x)}
\PgfmathsetmacroFPU{\myradius}{veclen(\pgf@y,\pgf@x)/1cm}
\PgfmathsetmacroFPU{\iter}{#1}
\PgfmathsetmacroFPU{\myx}{\myradius*cos((\iter)*\myphase)*0.8cm}% REAL PART
\PgfmathsetmacroFPU{\myy}{\myradius*sin((\iter)*\myphase)*0.8cm}% IMAGINARY PART
\pgf@x=\myx pt%
\pgf@y=\myy pt%
} 
\makeatother
